Based on the x86-64 architecture, this article begins with the execution mechanism of arithmetic logic instructions and provides an in-depth analysis of `leaq`, a specialized instruction capable of both memory addressing and efficient arithmetic optimization. We systematically organize the integer operation instruction set (covering the specialized register logic for 128-bit multiplication and division) and explain how condition codes capture operation side effects. Finally, through the `set` and `jump` instruction families, we demonstrate the implementation logic of low-level control flow.

## 1. Load Effective Address

### 1.1 lea/leaq

- **Basic Calculation Formula:**
  - `leaq Imm(Rb, Ri, S), %rax`
  - `%rax = Imm + Rb + Ri * S`
- **`mov` vs `leaq`:**
  - `mov` is used to access the memory at an address, while `leaq` is used to fetch the address itself.
  - `leaq (%rbx, %rcx), %rax` (# %rax = %rbx + %rcx is an address; does not access memory)
  - `movq (%rbx, %rcx), %rax` # %rax = M[%rbx + %rcx] (accesses memory)
- **Example:**
    ```asm
    # if M[0x111]=5 
    movq $0x110, %rbx
    movq $0x1, %rcx
    movq (%rbx, %rcx), %rax # %rax=5
    leaq (%rbx, %rcx), %rax # %rax=0x111
    # if don't know M[0x111]=?
    movq $0x110, %rbx
    movq $0x1, %rcx
    movq (%rbx, %rcx), %rax # %rax if definite but unknown
    ```

### 1.2 `leaq` Instruction Demonstration：
Let `%rbx=b`，`%rdx=d`
| Instruction | Effect |
|:---|:---|
| leaq 9(%rdx), %rax | %rax = 9+d |
| leaq (%rdx, %rbx), %rax | %rax = d+b |
| leaq (%rdx, %rbx, 3), %rax | %rax = d+3b |
| leaq 2(%rbx, %rbx, 7), %rax | %rax = 2+8b |
| leaq 0xE(, %rdx, 3), %rax | %rax = 14+3d |
| leaq 6(%rbx, %rdx, 7), %rax | %rax = 6+b+7d |

### 1.3 Arithmetic Scaling Example

```c
long scale(long x, long y, long z){
    long t = x + 4 * y + 12 * z;
    return t;
}
```

```asm
# x in %rdi, y in %rsi, z in %rdx 
scale:
    leaq (%rdi, %rsi, 4), %rax  # %rax = x + 4*y
    leaq (%rdx, %rdx, 2), %rdx  # %rdx = 3*z
    leaq (%rax, %rdx, 4), %rax  # %rax = x + 4*y + 12*z
```

## 2. Integer Operations

### 2.1 Overview of Common Arithmetic & Logic Instructions

In x86-64, arithmetic and logic instructions are generally categorized into unary and binary operations. A core constraint is: **two operands cannot both be memory addresses simultaneously.**

| Instruction | Effect | Description |
|:---|:---|:---|
| `leaq S, D` | D ← &S | Load Effective Address |
| `INC D` | D ← D + 1 | Increment |
| `DEC D` | D ← D − 1 | Decrement |
| `NEG D` | D ← −D | Negate |
| `NOT D` | D ← ∼D | Complement (Bitwise NOT) |
| `ADD S, D` | D ← D + S | Add |
| `SUB S, D` | D ← D − S | Subtract |
| `IMUL S, D` | D ← D × S | Integer Multiply |
| `XOR S, D` | D ← D ⊕ S | Exclusive-OR |
| `OR S, D` | D ← D ∣ S | Or |
| `AND S, D` | D ← D & S | And |

### 2.2 Examples: ASM $\leftrightarrow$ C Mapping

#### 2.2.1 Complex Arithmetic Transformation
```c
long arith(long x, long y, long z)
{
    long t1 = x ^ y
    long t2 = z * 48
    long t3 = t1 & 0x0F0F0F0F
    long t4 = t2 - t3
    return t4
}
```
```asm
# x in %rdi, y in %rsi, z in %rdx

arith:
    xorq %rsi, %rdi  # t1 = x ^ y
    leaq $48, %rdx, %rax  # t2 = z * 48
    andq $252645135, %rdi  # t3 = t1 & 0x0F0F0F0F
    subq %rdi, %rax # t4 = t2 - t3
    ret
```
#### 2.2.2
```asm
x in %rdi, y in %rsi, z in %rdx

arith:
    orq %rsi, %rdx
    sarq $9, %rdx
    notq %rdx
    movq %rdx, %rax
    subq %rsi, %rax
    ret
```
```c
short arith(short x, short y, short z)
{
    short p1 = y | z;
    short p2 = p1 >> 9;
    short p3 = ~p2;
    short p4 = p3 - y;
    return p4;
}
```

### 3. Special Arithmetic Operations

### 3.1 Overview

In x86-64, full precision multiplication and division utilize a register pair—`%rdx` and `%rax`—to form a contiguous 128-bit (oct-word) value area.

| Instruction | Effect | Description |
| :--- | :--- | :--- |
| `imulq S` | $R[\%rdx]:R[\%rax] \leftarrow S \times R[\%rax]$ | Signed full multiply (64-bit to 128-bit) |
| `mulq S` | $R[\%rdx]:R[\%rax] \leftarrow S \times R[\%rax]$ | Unsigned full multiply |
| `cqto` | $R[\%rdx]:R[\%rax] \leftarrow SignExtend(R[\%rax])$ | Sign-extend %rax into %rdx |
| `idivq S` | $R[\%rdx] \leftarrow \text{rem}, R[\%rax] \leftarrow \text{quot}$ | Signed divide ($RDX:RAX \div S$) |
| `divq S` | $R[\%rdx] \leftarrow \text{rem}, R[\%rax] \leftarrow \text{quot}$ | Unsigned divide ($RDX:RAX \div S$) |

### 3.2 Examples

#### 3.2.1 muliplication example

```c
void store_uprod(uint128_t *dest, uint128_t x, uint128_t y)
{
    *dest = x * (uint128_t) y;
}
```

```asm
# dest in %rdi, x in %rsi, y in %rdx

store_uprod:
    movq %rsi, %rax  # copy x
    mulq %rdx  # %rdx:%rax = y*x
    movq %rax, (%rdi)
    movq %rdx, 8(%rdi)
    ret
```

#### 3.2.2 division example

```c
void rediv(long x, long y, long *qp, long *rp)
{
    long q = x/y
    long r = x%y;
    *qp = q;
    *rp = r;
}
```

```asm
# x in %rdi, y in %rsi, qp in %rdx, rp in %rcx

rediv:
    movq %rdx, %r8  # copy qp
    movq %rdi, %rax  # move x to lower 8 bytes
    cqto  # sign-extend to upper 8 bytes
    idivq %rsi  # divide by y, %rdx = %rdx:%rax mod y(r), %rax = %rdx:%rax / y(q)
    movq %rax, (%r8)  # *qp = q
    movq %rdx, (%rcx)  # *rp = r
```

> Use `cqto` before `idivq`. 
> Manually use `movq $0, %rdx` before `divq`.

## 4. Control Flow

### 4.1 Condition Codes (Flags)

The CPU maintains a set of single-bit condition code registers that describe the attributes of the most recent arithmetic or logical operation. These flags are the physical basis for all branching decisions.

| Flag | Full Name | Description | Example |
| :--- | :--- | :--- | :--- |
| **CF** | Carry Flag | **Carry**: Unsigned overflow (carry/borrow from MSB). | `1111 + 0001` $\rightarrow$ `CF=1` |
| **ZF** | Zero Flag | **Zero**: Result of the recent operation was 0. | `subq %rax, %rax` $\rightarrow$ `ZF=1` |
| **SF** | Sign Flag | **Sign**: Result of the recent operation was negative (MSB=1). | `subq $10, $5` $\rightarrow$ `SF=1` |
| **OF** | Overflow Flag | **Overflow**: Signed 2's complement overflow (e.g., pos+pos=neg). | `p+p=n` $\rightarrow$ `OF=1` |

### 4.2 Dedicated instructions to alter condition codes (`cmp` & `test`)

These instructions only update the condition codes without altering the original data in the registers.

| Instruction | Based on Operation | Description |
| :--- | :--- | :--- |
| **`cmpq S1, S2`** | $S2 - S1$ | Compare: Behaves like `subq` but only sets flags. |
| **`testq S1, S2`** | $S1 \ \& \ S2$ | Test: Behaves like `andq` but only sets flags. |

### 4.3 `set` Instruction Family

`set` instructions assign 0 or 1 to a target operand (a low 8-bit register) based on the logical combination of condition codes.

| Instruction | Effect (Effect) | Description (Description) |
| :--- | :--- | :--- |
| `sete D` | $D \leftarrow ZF$ | equal / zero |
| `setne D` | $D \leftarrow \sim\!ZF$ | not equal / not zero |
| `sets D` | $D \leftarrow SF$ | negative |
| `setns D` | $D \leftarrow \sim\!SF$ | nonnegative |
| `setg D` | $D \leftarrow \sim\!(SF \oplus OF) \ \& \ \sim\!ZF$ | signed $>$ |
| `setge D` | $D \leftarrow \sim\!(SF \oplus OF)$ | signed $\ge$ |
| `setl D` | $D \leftarrow SF \oplus OF$ | signed $<$ |
| `setle D` | $D \leftarrow (SF \oplus OF) \mid ZF$ | signed $\le$ |
| `seta D` | $D \leftarrow \sim\!CF \ \& \ \sim\!ZF$ | unsigned $>$ |
| `setae D` | $D \leftarrow \sim\!CF$ | unsigned $\ge$ |
| `setb D` | $D \leftarrow CF$ | unsigned $<$ |
| `setbe D` | $D \leftarrow CF \mid ZF$ | unsigned $\le$ |

### 4.4 Jump Instructions

Jump instructions alter the program execution path by modifying the value of `%rip` based on the status of condition codes.

| Instruction | Jump Condition (Jump Condition) | Description (Description) |
| :--- | :--- | :--- |
| `jmp Label` | $1$ | direct jump |
| `jmp *Operand` | $1$ | indirect jump |
| `je Label` | $ZF$ | equal / zero |
| `jne Label` | $\sim\!ZF$ | not equal / not zero |
| `js Label` | $SF$ | negative |
| `jns Label` | $\sim\!SF$ | nonnegative |
| `jg Label` | $\sim\!(SF \oplus OF) \ \& \ \sim\!ZF$ | signed $>$ |
| `jge Label` | $\sim\!(SF \oplus OF)$ | signed $\ge$ |
| `jl Label` | $SF \oplus OF$ | signed $<$ |
| `jle Label` | $(SF \oplus OF) \mid ZF$ | signed $\le$ |
| `ja Label` | $\sim\!CF \ \& \ \sim\!ZF$ | unsigned $>$ |
| `jae Label` | $\sim\!CF$ | unsigned $\ge$ |
| `jb Label` | $CF$ | unsigned $<$ |
| `jbe Label` | $CF \mid ZF$ | unsigned $\le$ |

### 4.5 Example

```c
int comp(a, b)
```
```asm
a in %rdi, b in %rsi

comp:
    cmpq %rsi, %rdi  # c=%rdi-%rsi(a-b)
    setl %al  # if c<0(a<b), %al=1
    movzbl %al, %eax
    ret
```

## 5. Exercise

### 5.1 C to Assembly: Array Traversal and Condition Code Extraction

**C Source:** Determining if any negative numbers exist in an array.
```c
int negative_exists(int elems[], int N) {
    int res = 0;
    for (int i = 0; i < N; i++) {
        res |= (elems[i] < 0);
    }
    return res;
}
```
Assume `elems` base address in `%rdi`, N in `%esi`
```asm
negative_exists: 
    # Initialization
    movl $0, %eax           # %eax serves as res = 0
    movl $0, %ecx           # %ecx serves as loop variable i = 0
    cmpl %esi, %ecx         # Compare i and N
    jge .L3                 # If i >= N, jump directly to end (loop entrance guard)

.L1: # Loop Body
    movl (%rdi, %rcx, 4), %edx # Core addressing: %edx = M[%rdi + i*4] (i.e., elems[i])
    testl %edx, %edx        # Test the status of elems[i]
    sets %dl                # If result is negative (SF=1), set %dl = 1
    movzbl %dl, %edx        # Zero-extend: expand single-byte %dl to 32-bit %edx
    orl %edx, %eax          # res |= (elems[i] < 0)

.L2: # Loop Update 
    incl %ecx               # i++
    cmpl %esi, %ecx         # Compare i and N again
    jl .L1                  # If i < N, jump back to loop body

.L3:
    ret                     # Return res (%eax)
```

### 5.2 Reverse Engineering: Assembly $\rightarrow$ C

Assume `A` is in `%rdi`, `B` is in `%rsi`, and `N` is in `%edx`.

```asm
foo:
    pushq %rbp
    movq %rsp, %rbp
    movl $0, %eax           # a = 0
    movl $0, %ecx           # i = 0
    cmpl $0, %edx
    jle .L4                 # if N <= 0, goto end

.L1:
    movl (%rdi, %rcx, 4), %r8d  # %r8d = A[i]
    movl (%rsi, %rcx, 4), %r9d  # %r9d = B[i]
    cmpl %r9d, %r8d             # Compare A[i] and B[i]
    jg .L2                      # If A[i] > B[i] (signed greater than), jump to .L2
    addl %r9d, %eax             # Else (A[i] <= B[i]): a = a + B[i]
    jmp .L3                     # Skip .L2, enter loop update section

.L2:
    addl %r8d, %eax             # a = a + A[i]

.L3:
    incl %ecx                   # i++
    cmpl %edx, %ecx
    jl .L1                      # if i < N, continue loop

.L4:
    popq %rbp
    ret                         # return a
```

Restored C Source:
This assembly implementation follows a specific logic: it traverses two arrays and accumulates the larger of the two corresponding elements into a total sum.

```c
int foo(int *A, int *B, int N){
    int a = 0;
    for (int i = 0; i < N; i++) {
        if (A[i] > B[i]) {
            a = a + A[i];
        } else {
            a = a + B[i];
        }
    }
    return a;
}
```

### 5.3 The Cascade Effect of `movq` vs `leaq`
`leaq` calculates addresses, while `movq` fetches data.
Observe the following four code snippets; which one achieves `%rax = 15 * %rax` purely through arithmetic operations?

Snippet a: Double Memory Access
```asm
movq (%rax, %rax, 4), %rax  # %rax = M[5 * %rax]
movq (%rax, %rax, 2), %rax  # %rax = M[3 * M[5 * %rax]]
```
Snippet b: Memory Access followed by Arithmetic
```asm
movq (%rax, %rax, 4), %rax  # %rax = M[5 * %rax]
leaq (%rax, %rax, 2), %rax  # %rax = 3 * M[5 * %rax]
```
Snippet c: Arithmetic followed by Memory Access
```asm
leaq (%rax, %rax, 4), %rax  # %rax = 5 * %rax
movq (%rax, %rax, 2), %rax  # %rax = M[3 * (5 * %rax)] = M[15 * %rax]
```
Snippet d: Pure Arithmetic Accumulation
```asm
leaq (%rax, %rax, 4), %rax  # %rax = 5 * %rax
leaq (%rax, %rax, 2), %rax  # %rax = 3 * (5 * %rax) = 15 * %rax
```
Analysis:
Only `Snippet d` satisfies the condition. By cascading two `leaq` instructions, it never touches memory and utilizes only the ALU to complete the multiplication. This is a common technique used by compilers when optimizing constant multiplication.

### 5.4 Structures and Pointers: Memory Offsets

At the assembly level, structure (Struct) boundaries and field names disappear. The compiler calculates a fixed **Memory Offset** for each field based on the size of the data types. By analyzing the numeric values in the assembly code, we can accurately reverse-engineer the C language structure layout.

**C Structure Layout Mapping:**
```c
struct data {          // Total size: 28 bytes (ignoring end-of-struct padding)
    long id;           // Offset: 0  (takes 8 bytes)
    char surname[16];  // Offset: 8  (takes 16 bytes)
    int age;           // Offset: 24 (takes 4 bytes)
};

struct node {
    struct data d;     // Offset: 0  (takes 28 bytes)
    struct node *next; // Offset: 28 (takes 8 bytes)
};
```

**C Source Definitions:**
```c
short func_A(struct node *ptr) { 
    return ptr->d.id + 1; // Access starting member and perform addition
}
int *func_B(struct node *ptr) { 
    ptr = ptr->next;
    return &(ptr->d.age); // Jump through linked list and return member address
}
char func_C(struct node *ptr) { 
    return ptr->d.surname[9]; // Access specific byte offset
}
int *func_D(struct node *ptr) { 
    return ptr->d.age + ptr->next->d.age; // Combined operation on members across nodes
}
char *func_E(struct node *ptr) { 
    return &ptr->d.surname[16]; // Use address calculation to get array boundary pointer
}
```

**Corresponding ASM Functions:**
```asm
func_A:
    movq (%rdi), %rax    # Fetch ptr->d.id (Offset 0), using 'q' suffix for long
    addq $1, %rax        # Perform addition
    ret

func_B:
    movq 28(%rdi), %rdi  # Critical jump: ptr = ptr->next (load pointer at Offset 28)
    leaq 24(%rdi), %rax  # Calculate address: return age address at Offset 24 (no memory access)
    ret

func_C:
    movq 8(%rdi), %rax   # Access the starting point of data at Offset 8
    movq 9(%rax), %rax   # Perform second-level offset addressing
    movb (%rax), %al     # Extract single-byte result (1-byte)
    ret

func_D:
    movl 24(%rdi), %eax    # Read current node's age (Offset 24), 'l' suffix for int
    movq 28(%rdi), %rcx    # Load the next node's pointer (Offset 28)
    addl 24(%rcx), %eax    # Accumulate age field at Offset 24 of the next node
    ret

func_E:
    leaq 24(%rdi), %rax    # surname[0] is at Offset 8; adding index 16 equals Offset 24
    ret                    # Use leaq to return the physical address calculation result
```
